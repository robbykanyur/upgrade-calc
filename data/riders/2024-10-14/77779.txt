      function GraphDataService(endpoint, onsuccess) {
                
                $.ajax({
                    type: "GET",
                    contentType: "application/json; charset=utf-8",
                    url: endpoint,
                    async: true,
                    dataType: "json",
                    success: function (data, status) {
                        if(status == "success") {
                            if (onsuccess != undefined) {
                                onsuccess(data._data);
                            }
                        } else {
                            $("#graphtarget").hide();
                        }
                    },
                    error: function(data,status){
                        $("#graphtarget").hide();
                    }
                });
            }
     function updateChart(data){
              
              var total = 0;
              var compareTotal = 0;
              if(data.length > 1){
                  var graphData = new Array();
                  var cols = data.length;
                  
                  graphData[0] = new Array();
                  graphData[0][0] = 'Race';
                  graphData[0][1] = 'Points';                  
          
                  var totalraces = data.length;
                  var min = 9999;
                  var max = 0;
                  for(i = 0; i < data.length; i++)
                  {
                      graphData[i+1] = new Array();                           
                                                   
                      graphData[i+1][0] = data[i].X
                      graphData[i+1][1] = data[i].Y;                   

                      if(data[i].Y < min)
                        min = data[i].Y;
                      if(data[i].Y > max)
                        max = data[i].Y;

                  }
                   
                  if(max <= 1){
                    max = 1;
                    min = 0;
                  } else { 
                    min = min-50;
                  }

                  var options = {
                     
                      fontSize: 16,
                      //fontName: '"Open Sans", "Helvetica Neue", Helvetica, Arial, Geneva, sans-serif',
                      hAxis: { titleTextStyle: {color: '#333'}, minValue: 0},
                      vAxis: {minValue: min, maxValue: max,
                        viewWindow: { min: min, max: max}
                        },
                      legend: { position: "bottom" },
                      animation: {"startup": true,
                          duration: 1000,
                          easing: 'out'
                      }, 
                      chartArea: {
                          left: 40,
                          top: 20,
                          width: '95%'
                      },
                      series: {
                          0: { color: '#FC6719' },
                          1: { color: '#C4C4C4' }

    function PlaceMarker(map, marker,latLng, centerMap,color,caption) {
        if (marker != null) {
            marker.setMap(null);
        }
        if(centerMap)
            map.setCenter(latLng);
        marker = new google.maps.Marker({
            map: map,
            position: latLng,
            title: caption//,

            //icon: '/Images/orange.png'
        });
        if (caption)
            marker.setTitle(caption);
        if (color)
            marker.setIcon("http://maps.google.com/mapfiles/ms/icons/"+color+"-dot.png")
        return marker;
    }
